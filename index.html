デモ用の関数

import matplotlib.pyplot as plt
from pprint import pprint
import matplotlib.image as mpimg

def show(result):
    # pprint(result)
    pixelPerMM = result['pixelPerMM']
    w = result['width']
    h = result['height']
    print("1 mm = {} pixels".format(pixelPerMM))
    print("入力画像のサイズ: {} mm x {} mm".format(round(w / pixelPerMM), round(h / pixelPerMM)))
